`dockerfile`

```
# 기본 이미지 선택
FROM node:14

# 작업 디렉토리 설정
WORKDIR /app

# 의존성 파일 복사
COPY package.json yarn.lock ./

# 의존성 설치
RUN yarn install

# 애플리케이션 파일 복사
COPY . .

# 애플리케이션 실행을 위한 포트 열기
EXPOSE 4000

# 애플리케이션 실행 명령어
CMD ["node", "index.js"]

```

`### docker-compose.yml`
```
version: '3'
services:
  app:
    build: .
    ports:
      - "4000:4000"
    environment:
      NODE_ENV: production
  db:
    image: postgres:12
    environment:
      POSTGRES_USER: user
      POSTGRES_PASSWORD: password
    volumes:
      - db-data:/var/lib/postgresql/data
volumes:
  db-data:

```
